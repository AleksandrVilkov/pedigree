CREATE USER pedigree_user WITH password 'root';
ALTER USER pedigree SUPERUSER;
CREATE DATABASE pedigree OWNER pedigree;
CREATE TABLE PEDIGREE
(
    uid             varchar(80) PRIMARY KEY,
    createdDate     date NOT NULL,
    lastUpdatedDate date NOT NULL,
    ownerUid        varchar(80) NOT NULL
);

CREATE TABLE FAMILYMEMBERS
(
    uid                varchar(80) PRIMARY KEY,
    pedigreeUid        varchar(80) REFERENCES PEDIGREE(uid),
    createdDate        date NOT NULL,
    lastUpdatedDate    date NOT NULL,
    firstName          varchar(80) NOT NULL,
    lastName           varchar(80) NOT NULL,
    middleName         varchar(80),
    maidenName        varchar(80),
    dob                date NOT NULL,
    dod                date,
    mainPlaceResidence varchar(80),
    motherUid          varchar(80),
    fatherUid          varchar(80),
    hasChildren        bool,
    hasPartners        bool,
    note               varchar
);

CREATE TABLE CHILDREN_FAMILYMEMBER (
    familymemberUid varchar(80) REFERENCES FAMILYMEMBERS(uid) NOT NULL,
    childrenUid varchar(80) NOT NULL
);

CREATE TABLE PARTNER_FAMALYMEMBER (
    familymemberUid varchar(80) REFERENCES FAMILYMEMBERS(uid) NOT NULL,
    partnerUid varchar(80) NOT NULL
);
